# データベース設計

## テーブル構造

### receipts (レシート)
```sql
CREATE TABLE receipts (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID NOT NULL REFERENCES users(id),
  discord_user_id TEXT NOT NULL,
  image_url TEXT NOT NULL,
  thumbnail_url TEXT,
  ocr_raw_text TEXT,
  
  -- 抽出データ
  store_name TEXT,
  amount DECIMAL(10, 2),
  tax_amount DECIMAL(10, 2),
  receipt_date DATE,
  
  -- 分類
  category_id INTEGER REFERENCES expense_categories(id),
  category_confidence DECIMAL(3, 2), -- 0.00 ~ 1.00
  is_category_confirmed BOOLEAN DEFAULT FALSE,
  
  -- メタデータ
  notes TEXT,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  deleted_at TIMESTAMP WITH TIME ZONE
);

-- インデックス
CREATE INDEX idx_receipts_user_id ON receipts(user_id);
CREATE INDEX idx_receipts_discord_user_id ON receipts(discord_user_id);
CREATE INDEX idx_receipts_receipt_date ON receipts(receipt_date);
CREATE INDEX idx_receipts_category_id ON receipts(category_id);
```

### expense_categories (経費科目マスター)
```sql
CREATE TABLE expense_categories (
  id SERIAL PRIMARY KEY,
  code TEXT UNIQUE NOT NULL,
  name TEXT NOT NULL,
  description TEXT,
  parent_id INTEGER REFERENCES expense_categories(id),
  is_active BOOLEAN DEFAULT TRUE,
  sort_order INTEGER DEFAULT 0,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
```

### users (ユーザー)
```sql
CREATE TABLE users (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  discord_id TEXT UNIQUE NOT NULL,
  discord_username TEXT NOT NULL,
  email TEXT,
  default_category_id INTEGER REFERENCES expense_categories(id),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
```

### category_rules (自動分類ルール)
```sql
CREATE TABLE category_rules (
  id SERIAL PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  keyword TEXT NOT NULL,
  category_id INTEGER REFERENCES expense_categories(id),
  priority INTEGER DEFAULT 0,
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
```

### monthly_summaries (月次集計キャッシュ)
```sql
CREATE TABLE monthly_summaries (
  id SERIAL PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  year INTEGER NOT NULL,
  month INTEGER NOT NULL,
  category_id INTEGER REFERENCES expense_categories(id),
  receipt_count INTEGER DEFAULT 0,
  total_amount DECIMAL(12, 2) DEFAULT 0,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  UNIQUE(user_id, year, month, category_id)
);
```

## Supabase Storage バケット

```sql
-- レシート画像用バケット
INSERT INTO storage.buckets (id, name, public)
VALUES ('receipts', 'receipts', false);

-- ポリシー設定
CREATE POLICY "Users can upload their own receipts"
ON storage.objects FOR INSERT
WITH CHECK (auth.uid()::text = (storage.foldername(name))[1]);

CREATE POLICY "Users can view their own receipts"
ON storage.objects FOR SELECT
USING (auth.uid()::text = (storage.foldername(name))[1]);
```

## Row Level Security (RLS)

```sql
-- receipts テーブルの RLS
ALTER TABLE receipts ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can view own receipts"
ON receipts FOR SELECT
USING (user_id = auth.uid());

CREATE POLICY "Users can insert own receipts"
ON receipts FOR INSERT
WITH CHECK (user_id = auth.uid());

CREATE POLICY "Users can update own receipts"
ON receipts FOR UPDATE
USING (user_id = auth.uid());

-- monthly_summaries の RLS
ALTER TABLE monthly_summaries ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can view own summaries"
ON monthly_summaries FOR SELECT
USING (user_id = auth.uid());
```
